Route

Halaman Utama

Babar::route('/',function(){
   echo 'Hello World!';
});

Halaman Dengan Parameter URI Numerik
Contoh: localhost/babar/test/123 

Babar::route('/test/[:num]',function(){
   echo 'Ini halaman test uri numerik!';
});

Halaman Dengan Parameter URI String
Contoh: localhost/babar/test/satuduatiga 

Babar::route('/test/[:string]',function(){
   echo 'Ini halaman test uri string!';
});

Halaman Dengan Parameter URI Alpha Numerik
Contoh: localhost/babar/test/satudua3 

Babar::route('/test/[:alphanum]',function(){
   echo 'Ini halaman test uri alpha numerik!';
});

Halaman Dengan Parameter URI Wildcard
Contoh: localhost/babar/test/79315-71374 

Babar::route('/test/[:any]',function(){
   echo 'Ini halaman test uri wildcard!';
});

Halaman Dengan Parameter URI Gabungan
Contoh: localhost/babar/cek/test123/12345 

Babar::route('[:string]/[:alphanum]/[:num]',function(){
   echo 'Ini halaman test uri gabungan!';
});
