Helper

Json

1. Load Class

use system\helper\Json;

2. Json Encode

Json::encode('encode sekarang');

3. Json Decode

Json::decode($json_encoded_text);

Request

1. Load Class

use system\helper\Request;

2. Mendapatkan Parameter GET

Request::get(); // Return semua parameter GET (array)
Request::get('id'); // Return parameter GET dengan index 'id'

3. Mendapatkan Parameter POST

Request::post(); // Return semua parameter POST (array)
Request::post('id'); // Return parameter POST dengan index 'id'

4. Mendapatkan Parameter URI pada halaman yang sedang dibuka

Request::uri();

5. Mendapatkan Parameter URI pada halaman yang sedang dibuka berdasarkan segment

Request::uri_segment(); // Return semua URI Segment (array)
Request::uri_segment(1); // Return URI Segment pertama, segment dimulai dari 1

6. Mendapatkan Parameter Header

Request::header(); // Return semua parameter Header (array)
Request::header('Host'); // Return parameter Header dengan index 'Host'
7. Mendapatkan Parameter $_SERVER

Request::header(); // Return semua parameter $_SERVER (array)
Request::header('REQUEST_METHOD'); // Return parameter $_SERVER dengan index 'REQUEST_METHOD'
